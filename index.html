Drawing a line


import Svg
import Svg.Attributes as Attributes

line : Float -> Float -> Float -> Float -> Svg msg
line x1 y1 x2 y2 =
    Svg.line
        [ Attributes.x1 <| toString x1
        , Attributes.y1 <| toString y1
        , Attributes.x2 <| toString x2
        , Attributes.y2 <| toString y2
        ]
        []

translate : Float -> Float -> String
translate x y =
    " translate(" ++ toString x ++ "," ++ toString y ++ ") "

rotate : Float -> String
rotate degree =
    " rotate(" ++ toString degree ++ ") "

Making a grid


import List.Extra exposing (initialize, andThen)

makeGrid : Int -> Int -> List ( Float, Float )
makeGrid m n =
    initialize m toFloat
        |> andThen (\x -> initialize n toFloat
        |> andThen (\y -> [ ( x, y ) ]))

Recipe taken from Elm-Packages : List.Extra


> initialize 2 toFloat
[0,1] : List Float

> makeGrid 2 2

Generating random rotations

Elm Random Generators


generate : (a -> msg) -> Generator a -> Cmd msg
  • (a -> msg) — Msg constructor
  • Generator a — random generator
  • Cmd msg — Next Msg

Built-in package Random

Define our own generator


import Random exposing (Generator)

random : Generator Float
random =
    Random.float -0.5 0.5

randomList : Int -> Generator (List Float)
randomList n =
    Random.list n random

Generating random numbers


type Model
    = Setup Int Int Float               -- | Grid (m x n), and scale
    | Model Int Int Float (List Float)  -- | Mix in random values
          

type Msg
    = Generate
    | Draw (List Float)
          

init : ( Model, Cmd Msg )
init =
    update Generate (Setup 16 16 10)
          

update : Msg -> Model -> ( Model, Cmd Msg )
update msg model =
    case ( msg, model ) of
        ( Generate, Setup m n _) ) ->
            ( model, Random.generate Draw <| randomList (m * n) )

        ( Draw data, Setup m n size ) ->
            ( Model m n size data, Cmd.none )

        _ ->
            ( model, Cmd.none )
